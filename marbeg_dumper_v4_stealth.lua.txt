-- ─────────── 3. UTIL ───────────
local function fnv1a(s)
    if type(s) ~= "string" then s = tostring(s) end
    local h = 2166136261
    for i = 1, math.min(#s, 2048) do
        h = (h * 16777619) % 4294967296
        h = bit32.bxor(h, string.byte(s, i))
    end
    return string.format("%08x", h)
end

local function json_encode(v, depth)
    depth = depth or 0
    if depth > 6 then return '"<deep>"' end
    local t = type(v)
    if t == "nil" then return "null"
    elseif t == "boolean" then return v and "true" or "false"
    elseif t == "number" then
        if v ~= v or v == math.huge or v == -math.huge then return "null" end
        return tostring(v)
    elseif t == "string" then
        return '"' .. v:gsub('[\\"%c]', function(c)
            local b = string.byte(c)
            if b == 0x22 then return '\\"' elseif b == 0x5C then return '\\\\'
            elseif b == 0x0A then return '\\n' elseif b == 0x0D then return '\\r'
            elseif b == 0x09 then return '\\t' elseif b < 0x20 then return string.format('\\u%04x', b)
            else return c end
        end) .. '"'
    elseif t == "table" then
        local n, len = 0, #v
        for _ in pairs(v) do n = n + 1 end
        if len == n and n > 0 then
            local p = {} for i = 1, len do p[i] = json_encode(v[i], depth+1) end
            return "[" .. table.concat(p, ",") .. "]"
        else
            local p = {}
            for k, val in pairs(v) do
                p[#p+1] = json_encode(tostring(k), depth+1) .. ":" .. json_encode(val, depth+1)
            end
            return "{" .. table.concat(p, ",") .. "}"
        end
    end
    return '"<'..t..'>"'
end

local function write_json(path, data)

end

-- ─────────── 4. CORE CAPTURE ───────────
local function capture(origin, src, chunk_name)
    if type(src) ~= "string" or #src < 32 then return end
    if #src > CONFIG.max_capture_bytes then src = src:sub(1, CONFIG.max_capture_bytes) end
    local h = fnv1a(src)
    if state.dupe[h] then return end
    state.dupe[h] = true
    state.count = state.count + 1
    local n = state.count
    local fname = string.format("%03d_%s.lua", n, h)
    local info = {
        n=n, hash=h, origin=origin,
        chunk=tostring(chunk_name or ""):sub(1,80),
        size=#src, time=tick()-state.start,
    }
    state.captures[#state.captures+1] = info
    local hdr = string.format(
        "--[[ MARBEG v4 DUMP #%d\n  origin: %s\n  chunk:  %s\n  size:   %d\n  hash:   %s\n  time:   %.2fs\n]]--\n",
        n, origin, info.chunk, #src, h, info.time)
    pcall(_real.writefile, CONFIG.output_folder.."/"..fname, hdr..src)
    mark_dirty()
    state.last_activity = tick()
end

-- Build wrappers
local wrapped_loadstring, wrapped_load
wrapped_loadstring = function(src, chunkname)
    pcall(capture, "loadstring", src, chunkname)
    return _real.loadstring(src, chunkname)
end
wrapped_load = function(chunk, chunkname, mode, env)
    if type(chunk) == "string" then
        pcall(capture, "load", chunk, chunkname)
    elseif type(chunk) == "function" then
        -- bounded function-chunk capture
        local pieces, total, overflow = {}, 0, false
        local wrapped_fn = function()
            local p = chunk()
            if p and #p > 0 and not overflow then
                if total < CONFIG.max_capture_bytes and #pieces < 256 then
                    pieces[#pieces+1] = p; total = total + #p
                else overflow = true end
            end
            return p
        end
        local r, e = _real.load(wrapped_fn, chunkname, mode, env)
        if #pieces > 0 then
            local joined = table.concat(pieces)
            if overflow then joined = joined.."\n--[[ MARBEG v4: truncated ]]--" end
            pcall(capture, "load(fn)", joined, chunkname)
        end
        return r, e
    end
    return _real.load(chunk, chunkname, mode, env)
end

local function log_http(method, url, ok, resp_size)
    state.http_count = state.http_count + 1
    local item = {
        n = state.http_count, method = method,
        url = tostring(url):sub(1, 240),
        ok = ok, size = resp_size or 0,
        time = tick() - state.start,
    }
    state.http_log[#state.http_log+1] = item
    if #state.http_log > 200 then table.remove(state.http_log, 1) end
    mark_dirty()
    state.last_activity = tick()

    -- juga simpan body kalau response cukup besar untuk dianggap "chunk"
    return item
end

-- ─────────── 7. PUBLIC RUNNER ───────────
-- Cara aman jalankan Marbeg di sandbox.
-- Bisa terima: loader source, atau URL (akan diHttpGet), atau loader script
-- yang sudah membungkus loadstring(game:HttpGet(...))().
local function run_in_sandbox(src_or_url)
    if type(src_or_url) ~= "string" or #src_or_url == 0 then
        return false, "input kosong"
    end
    local src
    if src_or_url:match("^https?://") then
        local ok, resp = pcall(function() return game:HttpGet(src_or_url) end)
        if not ok then return false, "HttpGet gagal: "..tostring(resp):sub(1,80) end
        src = resp
        log_http("HttpGet(manual)", src_or_url, true, #src)
        pcall(capture, "manual:HttpGet", src, "@"..src_or_url:sub(1,60))
    else
        src = src_or_url
        pcall(capture, "manual:source", src, "<paste>")
    end
    local fn, err = _real.loadstring(src, "@marbeg_loader_sandbox")
    if not fn then return false, "compile: "..tostring(err):sub(1,100) end
    setfenv(fn, sandbox)
    local ok, run_err = pcall(fn)
    if not ok then
        state.last_error = "run:"..tostring(run_err):sub(1,80)
        return false, run_err
    end
    return true
end

                    last_origin = state.captures[#state.captures].origin
                end
                stats.Text = string.format(
                    "captures: %d   http: %d\nelapsed:  %.1fs\nlast:     %s (%d B)",
                    state.count, state.http_count, elapsed, last_origin, last_size)
                while seen < state.count do
                    seen = seen + 1
                    local cap = state.captures[seen]
                    if cap then
                        local row = Instance.new("TextLabel", history)
                        row.Size = UDim2.new(1,-8,0,14); row.BackgroundTransparency = 1
                        row.TextColor3 = Color3.fromRGB(180,230,180)
                        row.Font = Enum.Font.Code; row.TextSize = 10
                        row.TextXAlignment = Enum.TextXAlignment.Left
                        row.LayoutOrder = -seen
                        row.Text = string.format("#%d  %s  %dB  %.1fs",
                            cap.n, cap.origin:sub(1,14), cap.size, cap.time)
                    end
                end
                local rows = {}
                for _, c in ipairs(history:GetChildren()) do
                    if c:IsA("TextLabel") then rows[#rows+1] = c end
                end
                if #rows > CONFIG.max_ui_history then
                    table.sort(rows, function(a,b) return a.LayoutOrder > b.LayoutOrder end)
                    for i = CONFIG.max_ui_history+1, #rows do rows[i]:Destroy() end
                end
                history.CanvasSize = UDim2.new(0,0,0, hl.AbsoluteContentSize.Y + 4)
                local wait_sec = (tick() - state.last_activity < 3) and 0.5 or 2.0
                task.wait(wait_sec)
            end
        end)
    end)
    if not ok_ui then
        warn("[MARBEG v4] UI build error: "..tostring(err_ui))
        state.last_error = "UI:"..tostring(err_ui):sub(1,40)
    end
    return ok_ui
end
